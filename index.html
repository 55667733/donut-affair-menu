/* ----------------- CART LOGIC ----------------- */

let cart = {};


function formatRupee(n){ return '₹' + n; }


function createItemRow(item){

  const wrapper = document.createElement('div');

  wrapper.className = 'item-row';

  wrapper.innerHTML = `

    <div class="item-left">

      <div class="item-name">${item.name}</div>

    </div>

    <div style="display:flex;align-items:center">

      <div class="price">${formatRupee(item.price)}</div>

      <button class="add-btn" onclick="addToCart('${item.id}')">Add</button>

    </div>

  `;

  return wrapper;

}


function renderCategory(containerId, list){

  const container = document.getElementById(containerId);

  container.innerHTML = '';

  list.forEach(it => container.appendChild(createItemRow(it)));

}


function renderAllMenu(){

  renderCategory('combos', MENU.combos);

  renderCategory('packs', MENU.packs);

  renderCategory('additional', MENU.additional);

  renderCategory('coffee', MENU.coffee);

  renderCategory('tea', MENU.tea);

  renderCategory('shakes', MENU.shakes);

}


function addToCart(id){

  // find item across categories

  const all = [...MENU.combos, ...MENU.packs, ...MENU.additional, ...MENU.coffee, ...MENU.tea, ...MENU.shakes];

  const item = all.find(x=>x.id===id);

  if(!item) return;

  cart[id] = cart[id] || { ...item, qty:0 };

  cart[id].qty += 1;

  renderCart();

}


function removeFromCart(id){

  if(!cart[id]) return;

  cart[id].qty -= 1;

  if(cart[id].qty<=0) delete cart[id];

  renderCart();

}


function renderCart(){

  const list = document.getElementById('cartList');

  list.innerHTML = '';

  const keys = Object.keys(cart);

  if(keys.length===0){ list.innerHTML = '<div class="small">No items yet.</div>'; document.getElementById('totalPrice').innerText='₹0'; return; }


  let total = 0;

  keys.forEach(key=>{

    const it = cart[key];

    const subtotal = it.price * it.qty;

    total += subtotal;

    const row = document.createElement('div');

    row.className = 'cart-row';

    row.innerHTML = `

      <div style="max-width:65%">

        <div style="font-weight:700">${it.name}</div>

        <div class="small">Qty: ${it.qty} • ${formatRupee(it.price)} each</div>

      </div>

      <div style="text-align:right;min-width:140px">

        <div style="font-weight:800">${formatRupee(subtotal)}</div>

        <div style="margin-top:6px">

          <button class="add-btn" onclick="addToCart('${it.id}')">+</button>

          <button class="add-btn" style="background:#e33" onclick="removeFromCart('${it.id}')">-</button>

        </div>

      </div>

    `;

    list.appendChild(row);

  });

  document.getElementById('totalPrice').innerText = formatRupee(total);

}


function buildWhatsAppMessage(){

  const keys = Object.keys(cart);

  if(keys.length===0) return null;

  let msg = "Hello, I'd like to place an order from The Donut Affair:%0A";

  let total = 0;

  keys.forEach(k=>{

    const it = cart[k];

    const line = `- ${it.qty} x ${it.name} - ₹${it.price * it.qty}`;

    total += it.price * it.qty;

    msg += line + "%0A";

  });

  msg += `%0ATotal: ₹${total}%0A`;

  const nameRaw = document.getElementById('custName').value || '';

  const noteRaw = document.getElementById('custNote').value || '';

  if(nameRaw) msg += `%0AName: ${encodeURIComponent(nameRaw)}%0A`;

  if(noteRaw) msg += `%0ANotes: ${encodeURIComponent(noteRaw)}%0A`;

  msg += `%0AThanks!`;

  return msg;

}


document.getElementById('checkoutBtn').addEventListener('click', ()=>{

  const url = `https://wa.me/${+91 9152399943}?text=${Hi, I want to order}`;

  window.open(url, '_blank');

});
